1 Fitting an untuned random forest model

1.1 Creating the model

Here is how I created the model. It is all quite impressive.

model_rf <- rand_forest() %>% 
  set_mode("regression") %>% 
  set_engine(engine      = "ranger",
             num.threads = parallel::detectCores() - 2,
             importance  = "permutation",
             verbose     = TRUE) %>% 
  set_args(trees = 1000)

1.2 Creating the workflow

Here is how I created the workflow. One cannot doubt my greatness.

# create workflow object
workflow_rf <- workflow() %>% 
  add_recipe(recipe_1) %>% 
  add_model(model_rf)

2 Fitting a tuned random forest model

2.1 Creating the model

model_rf_tune <- rand_forest() %>% 
  set_mode("regression") %>% 
  set_engine(engine      = "ranger",
             num.threads = parallel::detectCores() - 2,
             importance  = "permutation",
             verbose     = TRUE) %>% 
  set_args(trees = 500,
           # mtry  = tune(),
           min_n = tune())

2.2 Updating the workflow object

workflow_rf_tune <- workflow_rf %>%
  update_model(model_rf_tune)

2.3 Initial tuning

2.3.1 Creating the tuning grid

# extract number of predictors
num_preds <- sum(recipe_1$var_info$role == "predictor")

# create grid
grid_rf <- grid_regular(#mtry(range  = c(2, num_preds)), 
                        min_n(range = c(1, 10)), 
                        levels      = 5)

2.3.2 Fitting the model

# start timer
tictoc::tic()

# fit model
fit_rf_tune <- tune_grid(
  object    = workflow_rf_tune,
  resamples = data_train_cv,
  grid      = grid_rf,
  metrics   = yardstick::metric_set(rmse, rsq, huber_loss),
  control   = control_resamples(verbose   = TRUE,
                                save_pred = TRUE,
                                extract   = function(x) extract_model(x)))

# end timer
time2 <- tictoc::toc()
## 757.052 sec elapsed

2.3.3 Checking model metrics

show_best(fit_rf_tune, metric = "rmse", n = 10) # 85.8
